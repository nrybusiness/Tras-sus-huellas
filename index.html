// ============================================ // TRAS SUS HUELLAS — Rakon Technology // Backend API v1.2 (con bienvenida emocional) // ============================================ const SHEET_ID = 'TU_SPREADSHEET_ID_AQUI'; const SHEETS = { PERDIDOS: 'Perdidos', AVISTAMIENTOS: 'Avistamientos', MATCHES: 'Matches' }; function doGet(e) { const action = e?.parameter?.action || ''; let result = { ok: false, error: 'Endpoint no soportado' }; try { switch (action) { case 'feed': const type = e.parameter.type; if (type !== 'perdidos' && type !== 'avistamientos') throw new Error('Tipo inválido'); result = { ok: true, data: getFeed(type) }; break; case 'stats': result = { ok: true, data: getStats() }; break; case 'bienvenida': result = { ok: true, data: getMensajeBienvenida() }; break; case 'setup': if (e.parameter.token === 'rakon2026') result = setupDatabase(); else result = { ok: false, error: 'No autorizado' }; break; default: result = { ok: false, error: 'Endpoint no soportado' }; } } catch (err) { result = { ok: false, error: err.toString() }; } return jsonResponse(result); } function doPost(e) { let result = { ok: false }; try { const payload = JSON.parse(e.postData.contents); if (payload.action === 'reportar') { const { type, data } = payload; const { id, matches } = crearReporte(type, data); result = { ok: true, id, matches }; } } catch (err) { result = { ok: false, error: err.toString() }; } return jsonResponse(result); } function getFeed(type) { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(type === 'perdidos' ? SHEETS.PERDIDOS : SHEETS.AVISTAMIENTOS); if (!sheet) return []; const data = sheet.getDataRange().getValues(); if (data.length <= 1) return []; const headers = data[0]; return data.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i] || ''); obj.type = type === 'perdidos' ? 'perdido' : 'avistamiento'; obj.estado = type === 'perdidos' ? (obj.estado || 'PERDIDO') : 'AVISTADO'; return obj; }).reverse(); } function crearReporte(type, data) { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(type === 'perdido' ? SHEETS.PERDIDOS : SHEETS.AVISTAMIENTOS); const lastRow = sheet.getLastRow(); let nextId = 1; if (lastRow > 1) { const lastId = sheet.getRange(lastRow, 1).getValue(); const num = lastId.match(/\d+/); if (num) nextId = parseInt(num[0]) + 1; } const id = (type === 'perdido' ? 'P-' : 'A-') + nextId.toString().padStart(4, '0'); let foto = data.foto || ''; if (foto.startsWith('data:image') && foto.length > 50000) { try { const blob = Utilities.newBlob(Utilities.base64Decode(foto.split(',')[1]), 'image/jpeg', id+'.jpg'); const file = DriveApp.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); foto = file.getUrl(); } catch(e) {} } sheet.appendRow([id, new Date(), data.nombre, data.especie, data.raza, data.color, data.tamaño, data.zona, data.descripcion, foto, data.contactoNombre, data.contactoTelefono, type==='perdido'?'PERDIDO':'AVISTADO']); const matches = findMatches({ id, type, ...data }); if (matches.length) { const matchSheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SHEETS.MATCHES); matches.forEach(m => matchSheet.appendRow(['M-'+Utilities.getUuid().slice(0,8), new Date(), type==='perdido'?id:m.id, type==='avistamiento'?id:m.id, m.score, false])); } return { id, matches }; } function findMatches(reporte) { const oppositeSheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(reporte.type === 'perdido' ? SHEETS.AVISTAMIENTOS : SHEETS.PERDIDOS); if (!oppositeSheet) return []; const data = oppositeSheet.getDataRange().getValues(); if (data.length <= 1) return []; const headers = data[0]; const candidates = data.slice(1).map(row => { const obj = {}; headers.forEach((h,i) => obj[h] = row[i]); return obj; }); return candidates.filter(c => c.especie === reporte.especie).map(c => { let score = 40; if (c.raza && reporte.raza && c.raza.toLowerCase() === reporte.raza.toLowerCase()) score += 25; else if (c.raza && reporte.raza && (c.raza.toLowerCase().includes(reporte.raza.toLowerCase()) || reporte.raza.toLowerCase().includes(c.raza.toLowerCase()))) score += 15; if (c.color === reporte.color) score += 15; if (c.tamaño === reporte.tamaño) score += 10; if (c.zona === reporte.zona) score += 10; return { id: c.id, score, pet: c }; }).filter(m => m.score >= 50).sort((a,b) => b.score - a.score).slice(0,5); } function getStats() { const ss = SpreadsheetApp.openById(SHEET_ID); const p = ss.getSheetByName(SHEETS.PERDIDOS), a = ss.getSheetByName(SHEETS.AVISTAMIENTOS); const totalP = p ? Math.max(0, p.getLastRow()-1) : 0; const totalA = a ? Math.max(0, a.getLastRow()-1) : 0; let resueltos = 0; if (p && totalP > 0) { const estados = p.getRange(2, 13, totalP).getValues().flat(); resueltos = estados.filter(e => e === 'ENCONTRADO').length; } return { totalPerdidos: totalP, totalAvistamientos: totalA, resueltos }; } function getMensajeBienvenida() { const stats = getStats(); const mensajes = [ `🐾 ${stats.resueltos} familias han vuelto a sonreír gracias a esta comunidad.`, `💚 Ya son ${stats.totalPerdidos + stats.totalAvistamientos} reportes que nos acercan a más reencuentros.`, `📍 Cubrimos 14 zonas del Valle de Aburrá. ¡Cada reporte cuenta!` ]; return { mensaje: mensajes[Math.floor(Math.random() * mensajes.length)], ...stats }; } function setupDatabase() { const ss = SpreadsheetApp.openById(SHEET_ID); const configs = [ { name: SHEETS.PERDIDOS, headers: ['id','timestamp','nombre','especie','raza','color','tamaño','zona','descripcion','foto','contactoNombre','contactoTelefono','estado'], color: '#E74C3C' }, { name: SHEETS.AVISTAMIENTOS, headers: ['id','timestamp','nombre','especie','raza','color','tamaño','zona','descripcion','foto','contactoNombre','contactoTelefono','matchedWith'], color: '#F1C40F' }, { name: SHEETS.MATCHES, headers: ['id','timestamp','perdidoId','avistamientoId','score','notificado'], color: '#1ABC9C' } ]; configs.forEach(cfg => { let sheet = ss.getSheetByName(cfg.name) || ss.insertSheet(cfg.name); let range = sheet.getRange(1,1,1,cfg.headers.length); if (range.getValues()[0].every(v => !v)) range.setValues([cfg.headers]); range.setFontWeight('bold'); sheet.setFrozenRows(1); sheet.autoResizeColumns(1, cfg.headers.length); sheet.setTabColor(cfg.color); }); return { ok: true, message: 'Base de datos configurada' }; } function jsonResponse(data) { return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); }
